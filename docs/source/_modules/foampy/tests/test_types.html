[docs]def test_foamlist():
    """Test `FoamList` class."""
    print("\nTesting FoamList")
    flist = FoamList([1, 2, 3, 4])
    print(flist)

    for i in flist:
        print(i)

    flist2 = FoamList([flist, flist], dtype=FoamList)
    print(flist2)

    flist = FoamList("(0 0 0 0)", dtype=int)
    assert flist[:] == [0]*4
